int pot = 0;
int int_led = 0;
void setup()
{
  pinMode(3, OUTPUT);
}

void loop()
{
  pot = analogRead(A0);
  int_led = map(pot, 0, 1023, 0, 255);
  analogWrite(3, int_led);
}
